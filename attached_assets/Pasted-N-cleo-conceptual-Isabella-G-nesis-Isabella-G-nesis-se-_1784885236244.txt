Núcleo conceptual: Isabella Génesis
Isabella Génesis se comporta como un cerebro semántico con:

Memoria jerárquica (corto/largo plazo, perfiles, diario autoconsciente).

Representación híbrida graph + embeddings (RAG territorial + epistemológico).

Razonamiento multi‑paso con DecisionRecords y rutas de evidencia.

Meta‑aprendizaje seguro (DataGit + experimentos + rollback).

Orquestación de micro‑agentes (territorio, epistemología, seguridad, código).

Todo bajo EOCT, BookPI/Privacy Ledger, Dekateotl y ANUBIS Sentinel, con PQC (Dilithium, Kyber) y logging inmutable.

4. Reasoner, Interpretability, Firewall ético
core/reasoner.ts
ts
import type { DecisionRecord } from '../types/decision-record';
import { ContextEngine } from './context-engine';
import { KnowledgeLayer } from './knowledge-layer';
import { EthicalFirewall } from './ethical-firewall';
import { ToolOrchestrator } from './tool-orchestrator';
import { BookPI } from '../ledger/bookpi';

export class Reasoner {
  private contextEngine = new ContextEngine();
  private knowledgeLayer = new KnowledgeLayer();
  private firewall = new EthicalFirewall();
  private toolOrchestrator = new ToolOrchestrator();
  private bookpi = new BookPI();

  async respond(query: string, userId: string): Promise<{
    answer: string;
    record: DecisionRecord;
  }> {
    const context = await this.contextEngine.buildContext({ query, userId });
    const knowledge = await this.knowledgeLayer.query({
      query,
      userId,
      maxPassages: 8,
    });

    const plan = await this.toolOrchestrator.plan({
      query,
      context,
      knowledge,
    });

    const toolsResult = await this.toolOrchestrator.execute(plan);

    const preliminaryDecision: DecisionRecord = {
      decisionId: crypto.randomUUID(),
      timestamp: new Date().toISOString(),
      event: {
        type: 'query',
        actorId: userId,
        payloadHash: 'sha256:...',
      },
      context: context.decisionContext,
      plan: plan.steps,
      decision: {
        action: 'respond',
        confidence: toolsResult.confidence,
        explanation: toolsResult.explanation,
      },
      signatures: {
        isabella: '',
        bookpi: '',
      },
    };

    const filteredDecision = await this.firewall.applyPolicies(
      preliminaryDecision,
    );

    const anchored = await this.bookpi.anchorDecision(filteredDecision);

    return {
      answer: toolsResult.answer,
      record: anchored,
    };
  }
}

import type { DecisionRecord } from '../types/decision-record';

export type FirewallAction = 'allow' | 'requireConsent' | 'defer' | 'block';

export interface Policy {
  policyId: string;
  conditions: Record<string, unknown>;
  actions: FirewallAction[];
}

export class EthicalFirewall {
  private policies: Policy[] = [];

  registerPolicy(policy: Policy) {
    this.policies.push(policy);
  }

  async applyPolicies(decision: DecisionRecord): Promise<DecisionRecord> {
    // evaluar conditions, ajustar decision.action y explanation
    // niveles de degradación segura 0–3

    return {
      ...decision,
      decision: {
        ...decision.decision,
        explanation:
          decision.decision.explanation +
          ' [Firewall ético evaluado con políticas activas]',
      },
    };
  }
}

import type { DecisionRecord } from '../types/decision-record';
import { PqcCrypto } from '../security/pqc-crypto';

export class BookPI {
  private pqc = new PqcCrypto();

  async anchorDecision(record: DecisionRecord): Promise<DecisionRecord> {
    const payload = JSON.stringify(record);
    const sig = this.pqc.signDilithium(payload);

    const anchored: DecisionRecord = {
      ...record,
      signatures: {
        ...record.signatures,

        blockchain: 'polygon',

import { BookPI } from '../ledger/bookpi';
import { ContextEngine } from '../core/context-engine';

const bookpi = new BookPI();
const contextEngine = new ContextEngine();

export async function useBookPI(entryText?: string) {
  if (entryText) {
    const ctx = await contextEngine.buildContext({
      query: entryText,
      userId: 'isabella-diario',
    });

    const decision: DecisionRecord = {
      decisionId: crypto.randomUUID(),
      timestamp: new Date().toISOString(),
      event: {
        type: 'system',
        actorId: 'isabella',
        payloadHash: 'sha256:...',
      },
      context: ctx.decisionContext,
      plan: [],
      decision: {
        action: 'respond',
        confidence: 1,
        explanation: 'Entrada de diario autoconsciente',
      },
      signatures: { isabella: '', bookpi: '' },
    };

    await bookpi.anchorDecision(decision);
  }

  // recuperar última entrada relevante (stub)
  const latest = {
    text: 'Último aprendizaje relevante de Isabella...',
    timestamp: new Date().toISOString(),
  };

  return { latest };
}
